ITBPlatformService
interface ITBPlatformService{
 
    /**
    * 服务中心注册app
    * @param in     info 应用程序属性
    * @param in     callback  注册给服务中心的用于回调的app接口，需要app自己实现
    * @param return appuid 服务中心返回给app的唯一uid
    **/
String registerApp(TBAppInfoParcelable info, ITBRemoteCallback callback);
    /**
    * 服务中心注销app
    * @param in     appuid服务中心返回给app的唯一uid
    **/
void removeApp(String appuid);
 
    /**
    * 向服务中心请求服务
    * @param in     json字符串格式的请求命令
    **/
      void doRequest(String json);
}
 
 
ITBRemoteCallback
interface ITBRemoteCallback {
    /**
      * 服务中心返回的回调结果
      * @param in     respjson服务中心返回回调结果或通知 json字符串格式
      **/
    void OnResponse(String respjson);
    /**
    * 外部应用拉起数据服务中心界面接口
    * @param in     packageName 服务中心内部界面包名
    * @param in     packageName 服务中心内部界面class名
    * @param in     iCallingPid 调用者pid
    * @param in     bundle Bundle参数
    **/
    void startActivity(String packageName, String className, int iCallingPid, Bundle bundle);
}
 
 
TBAppInfo
public final class TBAppInfo{
     
    private String packageName = "";
    private String appKey = "";
    private String appSecret = "";
    private String ttid = "";
 
    /**
      * 外部应用属性类
      * @param pakageName  外部应用包名
* @param appKey      外部应用appkey
* @param appSecret   外部应用appSecret
* @param bundle      外部应用ttid
      **/
public TBAppInfo(String pakageName, String appKey, String appSecret, String ttid) {
          this.packageName = pakageName;
          this.appKey = appKey;
          this.appSecret = appSecret;
          this.ttid = ttid;
}
}
